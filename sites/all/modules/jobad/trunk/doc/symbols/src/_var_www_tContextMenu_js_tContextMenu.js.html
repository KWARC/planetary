250       /**
251        * Returns the modules
252        * @returns {Object}  An object containing detailed description about dependencies
253        */
254       getModuleDependencies : function( module ){
255          if( typeof module == 'string' )
256             module = this._com.modules.disabled[module];
257          if( !module ){
258             console.warn( 'tContextMenu.checkModuleDependencies(): Invalid module');
259             return false;
260          }
261          
262          var result = {
263             modules  : {
264                enabled  : [],
265                disabled : [],
266                missing  : []
267             },
268             scripts  : {
269                loaded   : [],
270                missing  : []
271             }
272          };
273 
274          var d = module.info.dependencies;
275          for( var i in d ){
276             if( this._com.modules.enabled[d[i]] )
277                result.modules.enabled.push( d[i] );
278             else if( this._com.modules.disabled[d[i]] )
279                result.modules.disabled.push( d[i] );
280             else
281                result.modules.notFound.push( d[i] );
282          }
283          
284          for( var i in module.scripts ){
285             var script = this._com.componentContainer.find('script[src="'+module.scripts[i]+'"]');
286             if( script.length )
287                result.scripts.loaded.push( script );
288             else
289                result.scripts.missing.push( module.scripts[i] );
290          }
291          
292          return result;

295       /**
296        * _Attempts_ to resolve the dependencies of a module: loads missing scripts,
297        * enables disabled modules, fails upon missing module
298        * @returns {Boolean}
299        */
300       resolveModuleDependencies : function( module ){
301          if( typeof module == 'string' )
302             module = this._com.modules.disabled[module];
303          if( !module ){
304             console.warn( 'tContextMenu.resolveModuleDependencies(): Invalid module');
305             return false;
306          }
307          
308          var i, result = true;
309 
310          var d = this.getModuleDependencies( module );
311          if( d.modules.missing.length ){
312             console.warn( 'tContextMenu.resolveModuleDependencies(): Missing modules: '+d.modules.missing.implode(', ') );
313             result = false;
314          }
315          
316          if( d.modules.disabled.length ){
317             for( i in d.modules.disabled ){
318                if( !this.enableModule(d.modules.disabled[i]) ){
319                   console.warn( 'tContextMenu.resolveModuleDependencies(): ['+module.info.identifier+'] Unable to enable module: '+d.modules.disabled[i] );
320                   result = false;
321                }
322             }
323          }
324          
325          if( d.scripts.missing.length ){
326             for( i in d.scripts.missing ){
327                this.loadScript( d.scripts.missing[i] );
328             }
329          }
330          
331          return result;
332       },

336        * @param {String} path  The path to the script
337        * @param {String} identifier  An identifier for the script
338        * @returns {Boolean}  whether an apropriate tag was created or not
339        */
340       loadScript : function( path, identifier ){
341          identifier = identifier || 'script_'+Math.floor( Math.random()*696969 );
342          var script;
343          var extension = path.slice( path.lastIndexOf('.')+1 );
344          if( extension == 'css' ) {
345             script = $(document.createElement('link'))
346             script
347                .attr({
348                   'rel'    : 'alternate stylesheet',
349                   'type'   : 'text/css',
350                   'href'   : path,
351                   'target' : identifier,
352                   // Some browser require titles for style switching to work
353                   'title'  : identifier+'_'+Math.floor( Math.random()*696969 )
354                });
355          } else if( extension == 'js' ){
356             script = $(document.createElement('script'));
357             script
358                .attr({
359                   'type'   : 'text/javascript',
360                   'src'    : path,
361                   'target' : identifier
362                });
363          } else {
364             return false;
365          }
366          
367          this._com.componentContainer.append( script );
368          return script;

371       /**
372        * Dynamically and asynchronously load a module
373        *
374        * @param {String} path  The path to the file
375        * @param {Boolean} disabled  A boolean telling whether to have the module loaded but disabled or not
376        * @param {Function} callback  A callback function to be called after the load & initialization of the function
377                            but right before the 'afterModuleLoad' trigger. The callback and trigger only occur
378                            on success. For errors check the global jsonp error function and its triggers in 'init()'
379        */
380       loadModule  : function( path, callback, disabled, options ){
381          callback = callback || function(){};
382          
383          var com  = this._com;
384          var self = this;
385          
386          $(self).trigger('moduleLoad', [ path, callback, disabled, options ]);
387          
388          this.load(
389             path, 
390             function( component, path ){
391                var module = {};
392                $.extend( true, module, self.templates.module, component );
393                $.extend( module.options, options );
394                
395                if( com.modules.enabled[module.info.identifier] || com.modules.disabled[module.info.identifier] )
396                   return;
397                
398                com.modules.disabled[module.info.identifier] = module;
399                ++com.modules.disabled.__length;
400                if( !disabled )
401                   self.enableModule( module.info.identifier, options );
402 
403                callback( module, path, disabled, options );
404                $(self).trigger('afterModuleLoad', [ module, path, options, disabled ]);
405             }, 
406             {
407                callback : '_newModule'
408             }
409          );
410          
411          return this;
412       },

417        * @param {String} type  If not specified, it will return all modules: enabled + disabled
418        * @returns {Array}  A list of modules

420       getModules : function( type ){
421          if( typeof type == 'undefined' ){
422             return this.getModules('enabled').concat( this.getModules('disabled') );
423          } else {
424             var result = [];

427                   continue;
428                result.push( this._com.modules[type][i] );

434       /**
435        * Enabled a module
436        * @param {String} name  The module identifier
437        * @param {Object} options  The options for the module
438        * @returns {Boolean}  success/fail
439        */
440       enableModule : function( name, options ){
441          if( this._com.modules.enabled[name] )
442             return true;
443 
444          if( this._com.modules.disabled[name] ){
445             if( this.resolveModuleDependencies( name ) ){
446                this._com.modules.enabled[name] = this._com.modules.disabled[name];
447                delete this._com.modules.disabled[name];
448                this._com.modules.enabled[name].init( options );
449                ++this._com.modules.enabled.__length;
450                --this._com.modules.disabled.__length;
451                return true;
452             } else {
453                console.warn( 'tContextMenu.enableModule(): ['+name+'] Unable to resolve dependencies!' );
454                return false;
455             }
456          } else {
457             console.warn( 'tContextMenu.enableModule(): ['+name+'] Module not found!' );
458             return false;
459          }
460       },
